clear; clc; close all;

%% --- §0. USER SETTINGS -------------------------------------------------
DATA_DIR = '';   % <==== ENTER FOLDER PATH HERE

SAMPLES_PER_DAY = 576;
PRE_ROWS        = 576 * 5;
HIST_BINS       = 30;
SVM_C           = 10;
N_PERM          = 30;
SEED            = 42;
rng(SEED);

CHANNELS = {'Lat_diff','Lon_diff','Height_diff','ClockOffset', ...
            'TRODRY','TROWET','TGNTOT','TGETOT','TEC'};

% Unicode characters — renders correctly in all figure elements
D   = char(916);   % Delta
phi = char(966);   % phi
lam = char(955);   % lambda

CH_LABELS = {[D phi], [D lam], [D 'h'], ...
             'Clock Offset','Dry Troposphere','Wet Troposphere', ...
             'Tropo Gradient N','Tropo Gradient E','TEC'};
NC   = numel(CHANNELS);
DAYS = -5:0;

%% --- §1. EARTHQUAKE CATALOGUE ------------------------------------------
catalog(1).name = 'ECUADOR 2016';
catalog(1).file = fullfile(DATA_DIR, 'ECUADOR_2016_ALL.xlsx');
catalog(1).sta  = {'ESMR', 'QUI3'};

catalog(2).name = 'JAPAN 2011';
catalog(2).file = fullfile(DATA_DIR, 'JAPAN_2011_ALL.xlsx');

catalog(3).name = 'JAPAN 2024';

catalog(6).file = fullfile(DATA_DIR, 'CHILE_2010_ALL.xlsx');
catalog(6).sta  = {'ANTC', 'SANT'};

N_EQ = numel(catalog);

%% --- §2. DATA LOADING + FEATURE EXTRACTION -----------------------------
fprintf('=== §2: Data loading and feature extraction ===\n');

MAX_R  = 200;
MAX_S  = 20;

X_feat = nan(MAX_R, NC*4 + 1 + NC);
y_lbl  = nan(MAX_R, 1);
eq_ids = nan(MAX_R, 1);
day_vec= nan(MAX_R, 1);
sta_arr= cell(MAX_R, 1);
eq_arr = cell(MAX_R, 1);

SKLD_mat     = nan(MAX_S, 6, NC);
SKLD_eq_arr  = cell(MAX_S, 1);
SKLD_sta_arr = cell(MAX_S, 1);

feat_row = 0;
skld_row = 0;

for eq_i = 1:N_EQ
    eq = catalog(eq_i);
    fprintf('\n  [%d/%d] %s\n', eq_i, N_EQ, eq.name);

    for si = 1:numel(eq.sta)
        sta = eq.sta{si};

        try
            T = readtable(eq.file, 'Sheet', sta, 'VariableNamingRule', 'preserve');
        catch
            T = readtable(eq.file, 'Sheet', sta);
        end
        fprintf('    %s: %d rows\n', sta, height(T));

        pre_T = T(1:PRE_ROWS, :);
        eq_T  = T((PRE_ROWS+1):end, :);

        day_blocks = cell(6, 1);
        for d = 1:5
            day_blocks{d} = pre_T((d-1)*SAMPLES_PER_DAY+1 : d*SAMPLES_PER_DAY, :);
        end
        day_blocks{6} = eq_T;

        base_T = pre_T(1:(SAMPLES_PER_DAY*3), :);

        skld_row = skld_row + 1;
        SKLD_eq_arr{skld_row}  = eq.name;
        SKLD_sta_arr{skld_row} = sta;

        for d = 1:6
            blk       = day_blocks{d};
            day_label = DAYS(d);
            feat_row  = feat_row + 1;

            f_stat = nan(1, NC*4);
            for ci = 1:NC
                x = blk{:, CHANNELS{ci}};
                x = x(isfinite(x));
                if numel(x) >= 4
                    idx = (ci-1)*4;
                    f_stat(idx+1) = mean(x);
                    f_stat(idx+2) = std(x);
                    f_stat(idx+3) = skewness(x);
                    f_stat(idx+4) = kurtosis(x);
                end
            end

            skld_tec = compute_skld(blk{:,'TEC'}, base_T{:,'TEC'}, HIST_BINS);

            skld_ch = nan(1, NC);
            for ci = 1:NC
                skld_ch(ci) = compute_skld( ...
                    blk{:,CHANNELS{ci}}, base_T{:,CHANNELS{ci}}, HIST_BINS);
            end
            SKLD_mat(skld_row, d, :) = skld_ch;

            X_feat(feat_row, :) = [f_stat, skld_tec, skld_ch];
            y_lbl(feat_row)     = double(day_label >= -2);
            eq_ids(feat_row)    = eq_i;
            day_vec(feat_row)   = day_label;
            sta_arr{feat_row}   = sta;
            eq_arr{feat_row}    = eq.name;
        end
    end
end

valid   = ~isnan(y_lbl);
X_feat  = X_feat(valid, :);
y_lbl   = y_lbl(valid);
eq_ids  = eq_ids(valid);
day_vec = day_vec(valid);
sta_arr = sta_arr(valid);
eq_arr  = eq_arr(valid);

SKLD_mat    = SKLD_mat(1:skld_row, :, :);
SKLD_eq_arr = SKLD_eq_arr(1:skld_row);
SKLD_sta_arr= SKLD_sta_arr(1:skld_row);

X_feat(~isfinite(X_feat)) = 0;

% Feature names (internal, ASCII)
feat_names = {};
for ci = 1:NC
    feat_names{end+1} = [CHANNELS{ci} '_mean'];
    feat_names{end+1} = [CHANNELS{ci} '_std'];
    feat_names{end+1} = [CHANNELS{ci} '_skew'];
    feat_names{end+1} = [CHANNELS{ci} '_kurt'];
end
feat_names{end+1} = 'TEC_SKLD';
for ci = 1:NC
    feat_names{end+1} = ['SKLD_' CHANNELS{ci}];
end
N_FEAT = numel(feat_names);

% Display names: replace ENU codes with Unicode Greek symbols
% char(916)=Delta, char(966)=phi, char(955)=lambda
feat_disp = strrep(feat_names, 'Lat_diff',    [D phi]);
feat_disp = strrep(feat_disp,  'Lon_diff',    [D lam]);
feat_disp = strrep(feat_disp,  'Height_diff', [D 'h']);
feat_disp = strrep(feat_disp, '_', ' ');

fprintf('  Day-samples : %d\n', sum(valid));

    for ci = 1:NC
        baseline_ref(sri,ci) = nanmean(squeeze(SKLD_mat(sri,1:3,ci)));
    end

    n_sta    = sum(mask);

    ref_mean = nanmean(ref_sub,1);
    ref_std  = nanstd(ref_sub,0,1);

    for ci = 1:NC

             'Marker','o','MarkerSize',6, ...

    patch([DAYS(1) DAYS(end) DAYS(end) DAYS(1)], ...

    xlabel('Day (EQ = 0)','FontSize',10);

    title(sprintf('%s  (%d station(s))',catalog(eq_i).name,n_sta), ...

%% --- §4. INTERNAL EMPIRICAL THRESHOLDING --------------------------------
fprintf('\n=== §4: Empirical thresholding (Akyol §3.3) ===\n');
TAIL_ALPHA = 0.05;

fprintf('\n  Anomaly decisions (tail_p < %.2f):\n', TAIL_ALPHA);
fprintf('  %-22s %-10s  Day-2  Day-1  Day0\n','EQ/Station','Channel');

for sri = 1:skld_row
    for ci = 1:NC
        base_v = squeeze(SKLD_mat(sri,1:3,ci));
        base_v = base_v(isfinite(base_v));
        test_v = squeeze(SKLD_mat(sri,4:6,ci));

        tail_p = nan(3,1);
        for ti = 1:3
            if isfinite(test_v(ti)) && ~isempty(base_v)
                tail_p(ti) = sum(base_v > test_v(ti)) / numel(base_v);
            end
        end
        dec = tail_p < TAIL_ALPHA;

        anomaly_eq{end+1}     = SKLD_eq_arr{sri};
        anomaly_sta{end+1}    = SKLD_sta_arr{sri};
        anomaly_ch{end+1}     = CHANNELS{ci};
        anomaly_tail(end+1,:) = tail_p';
        anomaly_dec(end+1,:)  = dec';

            fprintf('  %-22s %-10s  %5d  %5d  %5d\n', ...
                [SKLD_eq_arr{sri} '/' SKLD_sta_arr{sri}], ...
                CHANNELS{ci}, dec(1), dec(2), dec(3));
        end
    end

    te = eq_ids == fold;

    X_te = X_feat(te,:); y_te = y_lbl(te);

    rng(SEED);
    svm_mdl = fitcsvm(X_tr, y_tr, ...
        'KernelFunction','rbf', ...
        'BoxConstraint', SVM_C, ...
        'KernelScale',   'auto', ...
        'Standardize',   true);

    try
        svm_mdl  = fitPosterior(svm_mdl);
        [~,sc]   = predict(svm_mdl, X_te);
        pos_col  = find(svm_mdl.ClassNames == 1);
        pos_sc   = sc(:, pos_col);
    catch
        [~,sc]   = predict(svm_mdl, X_te);
        pos_sc   = sc(:, end);
    end

    y_pred = predict(svm_mdl, X_te);

    [fpr_f,tpr_f,~,auc_f] = perfcurve(y_te, pos_sc, 1);
    fold_fpr{fold} = fpr_f;
    fold_tpr{fold} = tpr_f;
    fold_auc(fold) = auc_f;
    fold_mcc(fold) = compute_mcc(y_te, y_pred);

    for fi = 1:N_FEAT
        auc_p = nan(N_PERM,1);
        for rr = 1:N_PERM
            Xp = X_te;
            Xp(:,fi) = Xp(randperm(size(Xp,1)),fi);
            [~,sp] = predict(svm_mdl,Xp);
            try; ps=sp(:,pos_col); catch; ps=sp(:,end); end
            [~,~,~,auc_p(rr)] = perfcurve(y_te,ps,1);
        end
        fold_fi(fold,fi) = auc_f - nanmean(auc_p);
    end

    TP=sum(y_te==1&y_pred==1); FN=sum(y_te==1&y_pred~=1);
    FP=sum(y_te~=1&y_pred==1); TN=sum(y_te~=1&y_pred~=1);

        catalog(fold).name, auc_f, fold_mcc(fold), TP,FN,FP,TN);

%% ================================================================
%  LOCAL FUNCTIONS  
%% ================================================================

function d = compute_skld(x, y, n_bins)
    x = x(isfinite(x));
    y = y(isfinite(y));
    if numel(x)<2 || numel(y)<2; d=NaN; return; end
    lo = min([x;y]); hi = max([x;y]);
    if lo==hi; d=0; return; end
    edges = linspace(lo,hi,n_bins+1);
    ph = histcounts(x,edges,'Normalization','probability') + 1e-10;
    qh = histcounts(y,edges,'Normalization','probability') + 1e-10;
    ph = ph/sum(ph); qh = qh/sum(qh);
    d  = sum(ph.*log(ph./qh)) + sum(qh.*log(qh./ph));
end

function mcc = compute_mcc(y_true, y_pred)
    y_true = double(y_true==1);
    y_pred = double(y_pred==1);
    TP = sum(y_true==1 & y_pred==1);
    TN = sum(y_true==0 & y_pred==0);
    FP = sum(y_true==0 & y_pred==1);
    FN = sum(y_true==1 & y_pred==0);
    denom = sqrt((TP+FP)*(TP+FN)*(TN+FP)*(TN+FN));
    if denom==0; mcc=0; else; mcc=(TP*TN-FP*FN)/denom; end
end
